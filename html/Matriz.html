def initialize(matriz)
   @matriz = Array.new(matriz)
   @filas = matriz.size
   @columnas = matriz[0].size
   @n_elementos= (matriz.size * matriz[0].size)*0.6
    n_ceros=0
   filas.times do |i|
      columnas.times do |j|
      if (matriz[i][j]==0)
      n_ceros=n_ceros+1
                     end
   end
   end
   if n_ceros > @n_elementos
      raise RuntimeError, 'La Matriz no es densa'
   end
  
end

def maximo
   max=0;
   filas.times do |i|
      columnas.times do |j|
      if matriz[i][j] > max
      max=matriz[i][j]
      else
      end
      end
   end
   return max

def to_s

      0.upto(columnas-1) do |j|
          "#{matriz[i][j]} "
      end
